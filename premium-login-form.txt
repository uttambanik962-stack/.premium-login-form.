    <script>
        const loginForm = document.getElementById('loginForm');
        const emailInput = document.getElementById('email');
        const passInput = document.getElementById('password');
        const eyeBtn = document.getElementById('eyeball');
        const body = document.body;

        // Mode Toggle Logic
        eyeBtn.addEventListener('click', () => {
            body.classList.toggle('dark-mode');
            // Toggle password visibility
            passInput.type = passInput.type === 'password' ? 'text' : 'password';
        });

        // Form Submission Validation
        loginForm.addEventListener('submit', (e) => {
            e.preventDefault();
            let valid = true;

            [emailInput, passInput].forEach(input => {
                if (input.value.trim() === "") {
                    input.classList.add('error');
                    valid = false;
                } else {
                    input.classList.remove('error');
                }
            });

            if (valid) {
                alert("Login Successful! üöÄ");
            }
        });

        // Remove error highlight when typing
        emailInput.addEventListener('input', () => emailInput.classList.remove('error'));
        passInput.addEventListener('input', () => passInput.classList.remove('error'));
    </script>
